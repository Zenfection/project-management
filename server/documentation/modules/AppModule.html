import { Module } from '@nestjs/common';

import { IamModule } from './iam/iam.module';

@Module({
  imports: [
    PrismaModule.forRoot(),
    ConfigModule.forRoot(),
    IamModule,
    CloudModule,
    UsersModule,
  ],
  controllers: [AppController],
  providers: [AppService],
})
